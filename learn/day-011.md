L0083:     """Dynamically quantize a tensor to FP8 using tensorwise scaling.
L0084: 
L0085:     "Tensorwise" means one scalar scale for the entire tensor (as opposed to
L0086:     "rowwise" which computes a separate scale per row). Tensorwise is faster
L0087:     because cuBLAS handles the scaling; rowwise needs the CUTLASS kernel.
L0088: 
L0089:     Returns (fp8_data, inverse_scale) for use with torch._scaled_mm.
L0090:     """
L0091:     fp8_max = torch.finfo(fp8_dtype).max
L0092:     # Compute the max absolute value across the entire tensor
L0093:     amax = x.float().abs().max()
L0094:     # Scale maps [0, amax] -> [0, fp8_max]. Use float64 for the division to
L0095:     # ensure consistent numerics between torch.compile and eager mode.
L0096:     # (torchao does the same upcast — without it, compile/eager can diverge)
L0097:     scale = fp8_max / amax.double().clamp(min=EPS)
L0098:     scale = scale.float()
L0099:     # Quantize: scale into FP8 range, saturate (clamp prevents overflow when
L0100:     # casting — PyTorch's default is to wrap, not saturate), then cast to FP8

L0102:     x_clamped = x_scaled.clamp(-fp8_max, fp8_max)
L0103:     x_fp8 = x_clamped.to(fp8_dtype)
